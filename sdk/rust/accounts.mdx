# Account Wrappers

The Rust SDK provides type-safe wrappers for all Braid accounts.

## Market

```rust
use braid_client::Market;

let market = Market::new(&rpc, market_address);

// Fetch market account
let account = market.fetch()?;

// Access fields through account.data
account.data.admin
account.data.mint_a
account.data.mint_b
account.data.clmm_fee_bps
account.data.is_paused

// Or check if market exists first
let maybe_account = market.fetch_maybe()?;
if let Some(account) = maybe_account {
    println!("Market exists");
}
```

## User

```rust
use braid_client::User;

let user = User::new(&rpc, user_pda);
let account = user.fetch()?;

println!("Max orders: {}", account.data.max_orders);
println!("Current orders: {}", account.data.current_orders);
```

## Protocol

```rust
use braid_client::{Protocol, ProtocolPda, BRAID_ID};

let protocol_pda = ProtocolPda::derive_address(&BRAID_ID);
let protocol = Protocol::new(&rpc, protocol_pda);
let account = protocol.fetch()?;

println!("Admin: {}", account.data.admin);
```

## Shard Accounts

```rust
use braid_client::{ClmmOrders, LimitOrders, ClmmOrdersPda, LimitOrdersPda, BRAID_ID};

// Derive addresses
let clmm_address = ClmmOrdersPda::derive_address(&market, shard_idx, &BRAID_ID);
let limit_address = LimitOrdersPda::derive_address(&market, shard_idx, &BRAID_ID);

// Create wrappers
let clmm = ClmmOrders::new(&rpc, clmm_address);
let limit = LimitOrders::new(&rpc, limit_address);

// Fetch data
let clmm_account = clmm.fetch()?;
let limit_account = limit.fetch()?;
```
